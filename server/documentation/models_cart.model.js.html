/**
 * @module cart.model
 * @description Represents the cart schema for MongoDB.
 */

const mongoose = require("mongoose");

/**
 * @typedef {Object} CartItem
 * @property {string} productId - The ID of the product in the cart item.
 * @property {number} quantity - The quantity of the product in the cart item.
 */

/**
 * @typedef {Object} Cart
 * @property {string} _id - The unique ID of the cart.
 * @property {string} customerId - The ID of the customer associated with the cart.
 * @property {CartItem[]} items - The array of cart items.
 */

/**
 * @classdesc Represents the cart schema for MongoDB.
 * @class
 */
const cartSchema = new mongoose.Schema({
  /**
   * @member {string} Cart#customerId
   * @description The ID of the customer associated with the cart.
   */
  customerId: { type: mongoose.Schema.Types.ObjectId, ref: 'customers', unique: true },

  /**
   * @member {CartItem[]} Cart#items
   * @description The array of cart items.
   */
  items: [{
    /**
     * @member {string} CartItem#productId
     * @description The ID of the product in the cart item.
     */
    productId: { type: mongoose.Schema.Types.ObjectId, ref: 'products' },

    /**
     * @member {number} CartItem#quantity
     * @description The quantity of the product in the cart item.
     */
    quantity: Number
  }]
});

/**
 * @type {import('mongoose').Model<Cart>}
 */
module.exports = mongoose.model("carts", cartSchema);
