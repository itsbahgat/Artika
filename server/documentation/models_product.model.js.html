/**
 * @module product.model
 * @description Represents the product schema for MongoDB.
 */
const mongoose = require("mongoose");

/**
 * @typedef {Object} ProductReview
 * @property {string} user - The ID of the user who wrote the review.
 * @property {number} rating - The rating given by the user.
 * @property {string} comment - The comment provided by the user.
 * @property {Date} date - The timestamp of when the review was created.
 */

/**
 * @typedef {Object} Product
 * @property {string} _id - The unique ID of the product.
 * @property {string} title - The title of the product.
 * @property {string} description - The description of the product.
 * @property {number} price - The price of the product.
 * @property {string[]} categories - The categories the product belongs to.
 * @property {string[]} images - The URLs of the product images.
 * @property {string} seller - The ID of the user who is selling the product.
 * @property {Date} date - The timestamp of when the product was created.
 * @property {ProductReview[]} reviews - The array of product reviews.
 * @property {Date} createdAt - The timestamp of when the product was created.
 * @property {Date} updatedAt - The timestamp of when the product was last updated.
 */

/**
 * @classdesc Represents the product schema for MongoDB.
 * @class
 */
const productSchema = new mongoose.Schema({
    /**
     * @member {string} Product#title
     * @description The title of the product.
     */
    title: {
        type: String,
        required: true,
    },

    /**
     * @member {string} Product#description
     * @description The description of the product.
     */
    description: {
        type: String,
        required: true,
    },

    /**
     * @member {number} Product#price
     * @description The price of the product.
     */
    price: {
        type: Number,
        required: true,
    },

    /**
     * @member {string[]} Product#categories
     * @description The categories the product belongs to.
     */
    categories: [{
        type: String,
        required: true,
    }],

    /**
     * @member {string[]} Product#images
     * @description The URLs of the product images.
     */
    images: [{
        type: String,
        required: true,
    }],

    /**
     * @member {string} Product#seller
     * @description The ID of the user who is selling the product.
     */
    seller: {
        type: mongoose.Schema.Types.ObjectId,
        ref: 'User',
        required: true,
    },

    /**
     * @member {Date} Product#date
     * @description The timestamp of when the product was created.
     */
    date: {
        type: Date,
        default: Date.now,
    },

    /**
     * @member {ProductReview[]} Product#reviews
     * @description The array of product reviews.
     */
    reviews: [{
        /**
         * @member {string} ProductReview#user
         * @description The ID of the user who wrote the review.
         */
        user: {
            type: mongoose.Schema.Types.ObjectId,
            ref: 'User',
            required: true,
        },

        /**
         * @member {number} ProductReview#rating
         * @description The rating given by the user.
         */
        rating: {
            type: Number,
            required: true,
        },

        /**
         * @member {string} ProductReview#comment
         * @description The comment provided by the user.
         */
        comment: {
            type: String,
            required: false,
        },

        /**
         * @member {Date} ProductReview#date
         * @description The timestamp of when the review was created.
         */
        date: {
            type: Date,
            default: Date.now,
        },
    }],
}, {
    timestamps: true,
});

/**

@type {import('mongoose').Model<Product>}
*/
module.exports = mongoose.model("Product", productSchema);
