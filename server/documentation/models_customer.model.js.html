/**
 * @module customer.model
 * @description Represents the base user schema for customers in MongoDB.
 */

const mongoose = require('mongoose');
const bcrypt = require('bcrypt');

/**
 * @typedef {Object} Customer
 * @property {string} _id - The unique ID of the customer.
 * @property {string} avatar - The avatar of the customer.
 * @property {string} firstName - The first name of the customer.
 * @property {string} lastName - The last name of the customer.
 * @property {string} email - The email address of the customer.
 * @property {string} username - The username of the customer.
 * @property {string} password - The password of the customer.
 * @property {string} role - The role of the customer. Can be 'customer', 'seller', or 'admin'.
 * @property {string} address - The address of the customer.
 * @property {string} phone - The phone number of the customer.
 * @property {Date} createdAt - The timestamp of when the customer was created.
 * @property {Date} updatedAt - The timestamp of when the customer was last updated.
 */

/**
 * @classdesc Represents the base user schema for customers in MongoDB.
 * @class
 */
const baseUserSchema = new mongoose.Schema({
  /**
   * @member {string} Customer#avatar
   * @description The avatar of the customer.
   */
  avatar: String,

  /**
   * @member {string} Customer#firstName
   * @description The first name of the customer.
   */
  firstName: {
    type: String,
    required: true,
    trim: true
  },

  /**
   * @member {string} Customer#lastName
   * @description The last name of the customer.
   */
  lastName: {
    type: String,
    required: true,
    trim: true
  },

  /**
   * @member {string} Customer#email
   * @description The email address of the customer.
   */
  email: {
    type: String,
    required: true,
    trim: true,
    lowercase: true
  },

  /**
   * @member {string} Customer#username
   * @description The username of the customer.
   */
  username: {
    type: String,
    required: true,
    unique: true
  },

  /**
   * @member {string} Customer#password
   * @description The password of the customer.
   */
  password: {
    type: String,
    required: true
  },

  /**
   * @member {string} Customer#role
   * @description The role of the customer. Can be 'customer', 'seller', or 'admin'.
   */
  role: {
    type: String,
    required: true,
    enum: ['customer', 'seller', 'admin'],
    default: 'customer'
  },

  /**
   * @member {string} Customer#address
   * @description The address of the customer.
   */
  address: {
    type: String,
    required: true
  },

  /**
   * @member {string} Customer#phone
   * @description The phone number of the customer.
   */
  phone: {
    type: String,
    required: true
  }
},
{
  timestamps: true
});

/**
 * @function
 * @name baseUserSchema.pre('save')
 * @description Middleware function executed before saving a customer to encrypt the password.
 * @param {Function} next - The next function to be called.
 */
baseUserSchema.pre('save', async function (next) {
  let salt = await bcrypt.genSalt();
  this.password = await bcrypt.hash(this.password, salt);
next();
});

/**

@function
@name baseUserSchema.statics.login
@description Static method to log in a customer by email or username and password.
@param {string} emailOrUsername - The email or username of the customer.
@param {string} password - The password of the customer.
@returns {Promise<Customer>} The logged-in customer.
@throws {Error} If the login or password is incorrect.
*/
baseUserSchema.statics.login = async function (emailOrUsername, password) {
const user = await this.findOne({ $or: [{ email: emailOrUsername }, { username: emailOrUsername }] });
if (!user) throw new Error("Incorrect login or password");
const isPasswordValid = await bcrypt.compare(password, user.password);
if (!isPasswordValid) throw new Error("Incorrect login or password");

return user;
};

/**

@type {import('mongoose').Model<Customer>}
*/
module.exports = mongoose.model('Customer', baseUserSchema);
