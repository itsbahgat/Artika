/**
 * @module seller.model
 * @description Represents the seller schema for MongoDB.
 */

const mongoose = require('mongoose');
const user = require('./customer.model');

/**
 * @typedef {Object} SellerShop
 * @property {string} name - The name of the seller's shop.
 * @property {string} description - The description of the seller's shop.
 * @property {string} policies - The policies of the seller's shop.
 * @property {string[]} products - The IDs of the products associated with the seller's shop.
 */

/**
 * @typedef {Object} Seller
 * @property {string} _id - The unique ID of the seller.
 * @property {string} avatar - The avatar of the seller.
 * @property {string} firstName - The first name of the seller.
 * @property {string} lastName - The last name of the seller.
 * @property {string} email - The email of the seller.
 * @property {string} username - The username of the seller.
 * @property {string} password - The password of the seller.
 * @property {string} role - The role of the seller. Can be 'customer', 'seller', or 'admin'.
 * @property {string} address - The address of the seller.
 * @property {string} phone - The phone number of the seller.
 * @property {SellerShop} shop - The shop details of the seller.
 * @property {Date} createdAt - The timestamp of when the seller was created.
 * @property {Date} updatedAt - The timestamp of when the seller was last updated.
 */

/**
 * @classdesc Represents the seller schema for MongoDB.
 * @class
 */
const sellerSchema = new mongoose.Schema({
  /**
   * @member {string} Seller#avatar
   * @description The avatar of the seller.
   */
  avatar: String,

  /**
   * @member {string} Seller#firstName
   * @description The first name of the seller.
   */
  firstName: {
    type: String,
    required: true,
    trim: true,
  },

  /**
   * @member {string} Seller#lastName
   * @description The last name of the seller.
   */
  lastName: {
    type: String,
    required: true,
    trim: true,
  },

  /**
   * @member {string} Seller#email
   * @description The email of the seller.
   */
  email: {
    type: String,
    required: true,
    trim: true,
    lowercase: true,
  },

  /**
   * @member {string} Seller#username
   * @description The username of the seller.
   */
  username: {
    type: String,
    required: true,
    unique: true,
  },

  /**
   * @member {string} Seller#password
   * @description The password of the seller.
   */
  password: {
    type: String,
    required: true,
  },

  /**
   * @member {string} Seller#role
   * @description The role of the seller. Can be 'customer', 'seller', or 'admin'.
   */
  role: {
    type: String,
    required: true,
    enum: ['customer', 'seller', 'admin'],
    default: 'customer',
  },

  /**
   * @member {string} Seller#address
   * @description The address of the seller.
   */
  address: {
    type: String,
    required: true,
  },

  /**
   * @member {string} Seller#phone
 * @description The phone number of the seller.
   */
  phone: {
    type: String,
    required: true,
  },

  /**
   * @member {SellerShop} Seller#shop
   * @description The shop details of the seller.
   */
  shop: {
    name: String,
    description: String,
    policies: String,
    products: [{
      type: mongoose.Schema.Types.ObjectId,
      ref: 'Product',
    }],
  },
}, {
  timestamps: true,
});

/**
 * @type {import('mongoose').Model<Seller>}
 */
module.exports = mongoose.model('Seller', sellerSchema);
